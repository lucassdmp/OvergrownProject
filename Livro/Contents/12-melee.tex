\midsectiontitle{Combate Corpo a Corpo}

No campo de batalha de \textit{Overgrown}, a supremacia do combate corpo a corpo pertence àqueles que dominam não apenas a lâmina, mas a própria fúria que ela carrega. As \textbf{Habilidades de Combate} são técnicas ativas e instintivas que podem ser acionadas durante o combate gastando \textbf{Pontos de Combate (PC)}.

\begin{rpgboxtips}{Pontos de Combate}
Cada ponto em \textbf{Might} ou \textbf{Grace} concede 1 PC, mais 2 de base. Um personagem com 10 MIG e 10 GRA possui \textbf{22 PC}. Os PC se recuperam apenas com descanso longo. Habilidades na Árvore de Talentos podem aumentar o máximo ou permitir recuperação parcial.
\end{rpgboxtips}

\titlesection{Habilidades}

\combatSkill{Fúria}{10}{Você entra em um estado de \textit{Luta ou Fuga} instintivo. Seu personagem fica compelido a atacar quem considera inimigo, ganhando \textbf{mais um ataque por turno} e \textbf{5 de RD contra danos físicos} durante \textbf{1D6+1 rodadas}. Toda lógica cede ao instinto.}

\combatSkill{Fúria Descontrolada}{14}{Requer \textbf{Fúria ativa}. Ascende o estado de fúria ao limite absoluto: aumenta a duração em \textbf{+3 rodadas}, concede \textbf{mais um ataque por turno} e eleva a RD para \textbf{10 contra danos físicos e mágicos}. A cada rodada em que permanece ativo, você perde \textbf{1D4 de vida} pelo esforço extremo.}

\combatSkill{Pontos Fatais}{8}{Gaste sua \textbf{Reação} para realizar um teste de \link{per:discernimento}{Discernimento} contra a Percepção do alvo. Ao passar, você identifica uma abertura em suas defesas, reduzindo sua \textbf{RD apenas para os seus ataques}. Passar com até 10 remove \textbf{1 de RD}; passar com 11--15 remove \textbf{3 de RD}; passar com 16 ou mais remove \textbf{5 de RD}. Se a redução ultrapassar a RD total do alvo, você também ganha \textbf{+2 na Margem de Ameaça} contra esse alvo até o fim da cena.}

\combatSkill{Hemorragia Interna}{4}{Ao acertar um ataque em um inimigo que já está com \link{status:brutal}{Sangramento Brutal} ou \link{status:ardiloso}{Sangramento Ardiloso} ativo, você aprofunda o ferimento. O alvo sofre imediatamente \textbf{o dano do tique de sangramento atual como dano extra}, adicional ao dano normal do ataque.}

\combatSkill{Golpe Perfurante}{6}{Você direciona o golpe para vencer a proteção do inimigo. O ataque \textbf{ignora a Resistência da armadura} do alvo completamente, causando o dano base direto em seus pontos de vida. Não pode ser combinado com \textbf{Postura de Muralha}.}

\combatSkill{Golpe Duplo}{5}{Você desfere dois golpes rápidos no mesmo turno, sacrificando precisão por volume. Realiza dois ataques contra o mesmo alvo, cada um com \textbf{$-$2 de acerto}. Ambos podem resultar em crítico normalmente.}

\combatSkill{Garrote}{7}{O primeiro \textbf{Ataque Furtivo} bem-sucedido contra um alvo o deixa \link{status:silenciado}{Silenciado} por \textbf{3 turnos}, incapaz de conjurar magias ou emitir sons que pudessem alertar aliados. Pode ser usada sem estar oculto desde que o alvo esteja \link{status:desprevenido}{Desprevenido}.}

\combatSkill{Emboscar}{9}{O primeiro \textbf{Ataque Furtivo} bem-sucedido contra um alvo inflige \textbf{1,5$\times$ o dano normal}. A surpresa transforma o primeiro golpe em uma execução quase perfeita.}

\combatSkill{Provocação}{4}{Você instiga um inimigo inteligente a te atacar prioritariamente. Até o início do seu próximo turno, se o alvo atacar qualquer outro personagem que não seja você, sofre \textbf{Desvantagem} nesse ataque. Exige teste de \link{per:intimidacao}{Intimidação} contra a Temperança do alvo.}

\combatSkill{Pressão Constante}{8 por rodada}{Enquanto ativo, todo inimigo adjacente que tentar usar uma habilidade ou conjurar uma magia deve gastar \textbf{+2 PC adicionais} para fazê-lo. Você domina o espaço físico ao redor, tornando cada decisão do inimigo mais custosa. Cancele como ação livre.}

\combatSkill{Marcar para a Morte}{14}{Você foca toda sua letalidade em um único alvo. Ganha \textbf{+3 no acerto}, \textbf{+2 na Margem de Ameaça} e \textbf{+1D6 de dano} contra ele. Em contrapartida, o dano recebido de todas as outras fontes aumenta em 25\% e você não pode \textbf{desviar ou resistir} ao que não vem do alvo marcado. \par\textit{\footnotesize \textbf{Aprimorado:} As penalidades são progressivamente removidas conforme a habilidade é refinada na Árvore de Talentos.}}

\combatSkill{Execução}{12}{Se o alvo estiver com \textbf{25\% de vida ou menos}, o ataque tem \textbf{Vantagem} e, ao acertar, causa o \textbf{dobro do dano}. Não pode ser usada contra alvos que já estejam na condição \link{status:morrendo}{Morrendo}.}

\combatSkill{Última Resistência}{20}{Ao receber um dano que seria fatal, ao invés de entrar em \link{status:morrendo}{Morrendo}, você permanece com \textbf{1 de vida} e zera seus PC. Role \textbf{1D4}: durante esse número de turnos você está \textbf{imune à condição Morrendo}. Ao fim desse período, se ainda estiver sem cura, entra em Morrendo normalmente. Só pode ser ativada uma vez por combate.}

\combatSkill{Escudo Humano}{7}{Use sua \textbf{Reação} para empurrar um aliado adjacente para fora da trajetória de um ataque direcionado a ele. Você absorve \textbf{metade do dano} no lugar do aliado. Deve ser declarado antes do resultado do ataque inimigo.}

\combatSkill{Vício em Sangue}{6}{Se um alvo com \textbf{metade da vida ou menos} estiver em alcance médio, você se arremessa até ele \textbf{sem gastar sua Ação de Movimento}. O cheiro de sangue aguça o instinto predatório.}

\combatSkill{Sede de Sangue}{10}{Ao \textbf{executar} um inimigo (reduzir sua vida a 0), você canaliza o ímpeto da matança em vitalidade. Cure uma quantidade de pontos de vida igual a \textbf{metade do dano do seu último ataque} contra o alvo abatido.}

\combatSkill{Instinto Sanguinário}{13}{Você abandona a capacidade de \textbf{Reagir} em troca de instinto puro. Para cada participante do combate (aliado ou inimigo) com menos da metade da vida, você causa \textbf{+1D4 de dano} em todos os seus ataques. O bônus máximo acumulável é \textbf{+5D4}, independentemente do número de combatentes.}

\combatSkill{Elipse Carmesim}{10 / 15}{Você gira e corta tudo ao seu redor: causa o \textbf{dano da arma em todos os inimigos} em alcance curto que estejam com \link{status:ardiloso}{Sangramento Ardiloso} ou \link{status:brutal}{Sangramento Brutal} ativo. \par\textit{\footnotesize \textbf{Aprimorado (15 PC):} O ataque pode resultar em \textbf{crítico em cada inimigo acertado}, não apenas no alvo principal.}}

\combatSkill{Desarmar}{7}{Tente arrancar a arma do alvo em pleno combate. Faça uma \textbf{disputa de testes}: seu \textbf{Might} contra o \textbf{Might} do alvo. Se vencer, a arma cai no terreno adjacente e o alvo fica \link{status:desprevenido}{Desprevenido} por 1 turno. Alvos sem arma empunhada são imunes a esta habilidade.}

\combatSkill{Ataque Secundário}{5}{Gaste PC para realizar \textbf{um segundo ataque} contra o mesmo alvo no mesmo turno, como uma ação adicional dentro da sua Ação Padrão.}

\combatSkill{Desarranjar}{6}{Realiza uma manobra de afastamento que nega \textbf{ataques de oportunidade} ao se reposicionar, quebrando o engajamento sem expor a guarda.}

\combatSkill{Pisada de Impacto}{11}{Requer \textbf{Fúria ativa}. Você golpeia o chão com força brutal, causando uma onda de choque em cone de 3 unidades à sua frente. Todos os alvos na área sofrem \textbf{1D6 + MOD de Might} de dano e devem realizar um teste de Fortitude (DT 14) ou receber \link{status:derrubado}{Derrubado}.}

\combatSkill{Controle de Zona}{10}{Até o início do seu próximo turno, declare uma área de 2 unidades ao redor de você como \textbf{Zona Controlada}. Qualquer inimigo que tentar entrar, sair ou se mover dentro dela sofre um \textbf{ataque de oportunidade gratuito} de sua parte. Você pode cancelar a habilidade como ação livre.}

\combatSkill{Golpe Atordoante}{9}{Você direciona o impacto aos sentidos do alvo. Em vez de causar dano normal, o ataque aplica \link{status:concussionado}{Concussão} automaticamente, sem necessidade de acerto crítico. O alvo ainda pode testar \link{att:fortitude}{Fortitude} para resistir, com DT igual ao resultado do seu ataque.}

\titlesection{Mecânicas de Esforço}

As Mecânicas de Esforço são manobras e técnicas táticas disponíveis a qualquer combatente. Diferente das Habilidades, não exigem especialização, apenas PC e o momento certo.

\begin{rpgboxtips}{Quando Usar}
As mecânicas de esforço são opções abertas para todos os personagens, independente de estilo. Combiná-las com as habilidades certas pode transformar uma troca de golpes comum em uma vitória dominante.
\end{rpgboxtips}

\combatSkill{Agarrar}{7}{Realiza uma manobra de imobilização. Faça uma \textbf{disputa de testes}: seu \textbf{Might} contra o \textbf{Grace} do alvo. Se vencer, o alvo fica \link{status:segurado}{Segurado}: não pode se mover e sofre \textbf{Desvantagem} em ataques. Para se libertar, o alvo deve vencer uma nova disputa de \textbf{Might vs. Might} no início de cada um de seus turnos, ou ser auxiliado por um aliado.}

\combatSkill{Impulsão}{4}{Gaste PC para ganhar \textbf{+3 unidades} na sua Ação de Movimento neste turno, canalizando um burst de energia física para cobrir mais terreno.}

\combatSkill{Contra-Ataque}{7}{Se um inimigo \textbf{errar um ataque} contra você, use sua \textbf{Reação} para responder imediatamente com um ataque contra ele, sem gastar sua Ação Padrão. \textbf{Não pode ser usado} em conjunto com Esquiva: se você esquivou do ataque (gastando sua Reação), o Contra-Ataque não está disponível naquele mesmo ataque.}

\combatSkill{Recuar Estratégico}{6}{Ao desengajar, use esta técnica para se afastar o \textbf{dobro da distância} normal, criando uma zona de segurança sem provocar ataques de oportunidade.}

\combatSkill{Inabalável}{6}{Use como \textbf{Reação} para prevenir as condições \link{status:atordoado}{Atordoado} ou \link{status:derrubado}{Derrubado} antes que entrem em efeito. Deve ser declarado antes do resultado.}

\combatSkill{Estocada Interruptora}{9}{Use sua \textbf{Reação} para interromper uma habilidade ou magia em execução pelo inimigo. Se o ataque acertar, o inimigo gasta a ação sem efeito algum.}

\combatSkill{Comando de Assalto}{7}{Gaste sua \textbf{Ação de Movimento} para emitir uma ordem tática a um aliado. Ele realiza \textbf{um ataque fora do seu turno} contra o inimigo de sua escolha, usando seus próprios atributos.}

\combatSkill{Analisar Padrão}{7}{Gaste sua \textbf{Ação de Movimento} estudando os movimentos do inimigo. Todos os seus ataques contra ele neste turno têm \textbf{Vantagem}. Se não atacar neste turno, o bônus se transfere para o turno seguinte.}

\combatSkill{Postura de Muralha}{4 por rodada}{Você não pode atacar neste turno, mas \textbf{dobra sua Resistência} até o início do seu próximo turno. Cada aliado adjacente ganha \textbf{+2 de Resistência} enquanto você mantiver a postura.}

\combatSkill{Rasteira}{7}{Tente derrubar o alvo aplicando \link{status:derrubado}{Derrubado}. Teste de \textbf{Grace} do alvo contra o seu \textbf{teste de Might}. Se falhar, o alvo cai. Alvos \link{status:derrubado}{Derrubados} sofrem \textbf{Vantagem} em ataques corpo a corpo recebidos até se levantarem.}
